### Component usage

```js

import { Dropdown, Option } from '@/vhs-ui/elements/Dropdown'

const options = [
  { label: 'Pick 1', value: 'pick1', image: NamePlate },
  { label: 'Pick 2', value: 'pick2', image: NamePlate },
  { label: 'Pick 3', value: 'pick3', image: NamePlate },
]

export const DropdownExample = () => {
  const [selected, setSelected] = useState<Option>()

  return (
      <Dropdown size="md" onSelection={setSelected} selected={selected}>
        {options}
      </Dropdown>
  )
}
```
