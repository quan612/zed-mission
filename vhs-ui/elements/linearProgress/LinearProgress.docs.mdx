### Component usage

```js
<LinearProgress max={10} current={3}>
  {({ stepsArray }) => (
    <>
      {stepsArray.map(({ value, isActive, isFirst, isInProg }) => (
        <LinearProgressStep
          key={value}
          value={value}
          isActive={isActive}
          isFirst={isFirst}
          isInProg={isInProg}
        />
      ))}
    </>
  )}
</LinearProgress>
```
